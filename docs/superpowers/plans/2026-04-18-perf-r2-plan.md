```rust
// ---- pprof-rs profiling hook (R2 diagnosis) ----
#[cfg(feature = "profiling")]
{
    if let Ok(secs_s) = std::env::var("AUTUMN_PPROF_SECS") {
        if let Ok(secs) = secs_s.parse::<u64>() {
            if secs > 0 {
                let out_path = std::env::var("AUTUMN_PPROF_OUT")
                    .unwrap_or_else(|_| "/tmp/autumn_ps_pprof.svg".to_string());
                let thread_filter = std::env::var("AUTUMN_PPROF_THREADS").ok();
                std::thread::spawn(move || {
                    let guard = pprof::ProfilerGuardBuilder::default()
                        .frequency(99)
                        .blocklist(&["libc", "libgcc", "pthread", "vdso"])
                        .build()
                        .expect("pprof guard");
                    std::thread::sleep(std::time::Duration::from_secs(secs));
                    let report = guard.report().build().expect("pprof report");
                    let mut file = std::fs::File::create(&out_path).expect("pprof outfile");
                    // Optional per-thread-name filter (AUTUMN_PPROF_THREADS="part-13" or "ps-conn-")
                    if let Some(prefix) = thread_filter {
                        use pprof::protos::Message;
                        // Build an SVG with only frames whose thread_name starts with the prefix.
                        // pprof::Report doesn't expose thread filtering directly, so write the
                        // full SVG and also write a prefix-filtered text summary alongside.
                        report.flamegraph(&mut file).expect("flamegraph write");
                        let txt_path = format!("{}.threads.txt", out_path);
                        let mut txt = std::fs::File::create(&txt_path).expect("thread summary");
                        for frames in report.data.keys() {
                            if frames.thread_name.starts_with(&prefix) {
                                use std::io::Write;
                                writeln!(
                                    txt,
                                    "thread={} count={}",
                                    frames.thread_name,
                                    report.data.get(frames).copied().unwrap_or(0)
                                ).ok();
                            }
                        }
                    } else {
                        report.flamegraph(&mut file).expect("flamegraph write");
                    }
                    eprintln!("[R2] pprof flamegraph written: {}", out_path);
                });
            }
        }
    }
}
// ---- end pprof hook ----
```

Place this inside `main()` as early as possible (right after the `tracing_subscriber` init), so the sampler starts before any heavy allocations. The hook spawns its own `std::thread` (not a compio task) so it doesn't steal from the event loop.

```python
#!/usr/bin/env python3
"""Per-thread CPU sampler for an autumn-ps PID.

Usage: perf_r2_thread_cpu.py <pid> [--duration 2] [--top 15]

Samples /proc/<pid>/task/*/stat at t=0 and t=duration, reports per-thread
CPU percentage, and prints both an aggregate-by-comm table and top-N
individual threads. Designed to run alongside a pprof capture so the
thread-name distribution is recorded in the same window.
"""

import os
import sys
import time
from collections import defaultdict


def _snap(pid: str) -> dict:
    out = {}
    for tid in os.listdir(f"/proc/{pid}/task"):
        try:
            stat = open(f"/proc/{pid}/task/{tid}/stat").read()
            comm = open(f"/proc/{pid}/task/{tid}/comm").read().strip()
        except FileNotFoundError:
            continue
        # /proc/.../stat has (comm) which may contain spaces; split after the closing paren.
        parts = stat.rsplit(")", 1)[1].split()
        utime = int(parts[11])
        stime = int(parts[12])
        out[tid] = (comm, utime + stime)
    return out


def main() -> None:
    import argparse

    parser = argparse.ArgumentParser()
    parser.add_argument("pid")
    parser.add_argument("--duration", type=float, default=2.0)
    parser.add_argument("--top", type=int, default=15)
    args = parser.parse_args()

    hz = int(os.sysconf(os.sysconf_names["SC_CLK_TCK"]))  # usually 100
    s1 = _snap(args.pid)
    time.sleep(args.duration)
    s2 = _snap(args.pid)

    cpus = defaultdict(list)
    for tid, (comm, t2) in s2.items():
        if tid not in s1:
            continue
        _, t1 = s1[tid]
        dt = (t2 - t1) / hz
        pct = dt * 100 / args.duration
        cpus[comm].append((tid, pct))

    aggs = []
    for comm, lst in cpus.items():
        total = sum(p for _, p in lst)
        mx = max(p for _, p in lst)
        aggs.append((comm, len(lst), total, mx))
    aggs.sort(key=lambda x: -x[2])

    print(f"{'thread_name':<20} {'N':>4}  {'total %':>8}  {'max %':>6}")
    print("-" * 45)
    for comm, n, total, mx in aggs[:args.top]:
        print(f"{comm:<20} {n:>4}  {total:>8.1f}  {mx:>6.1f}")

    flat = [(pct, comm, tid) for comm, lst in cpus.items() for tid, pct in lst]
    flat.sort(reverse=True)
    print()
    print(f"Top {args.top} busiest individual threads:")
    for pct, comm, tid in flat[: args.top]:
        print(f"  tid={tid:<8} {comm:<22} {pct:>6.1f}%")

    grand = sum(total for _, _, total, _ in aggs)
    print()
    print(f"Total process CPU (sum of thread %): {grand:.1f}%")


if __name__ == "__main__":
    main()
```

```bash
#!/usr/bin/env bash
# perf_r2_flamegraph.sh — capture the 4 R2 flamegraphs with consistent setup.
#
# Runs each capture sequentially:
#   C1  --shm N=1  full PS    15s    perf_r2_ps_n1_full.svg
#   C2  --shm N=1  P-log only 15s    perf_r2_ps_n1_plog.svg  (via AUTUMN_PPROF_THREADS=part-)
#   C3  --shm N=1  ps-conn-*  15s    perf_r2_ps_n1_conn.svg  (via AUTUMN_PPROF_THREADS=ps-conn)
#   C4  --shm N=4  full PS    15s    perf_r2_ps_n4_full.svg
#
# Outputs written to $OUT_DIR (default: scripts/perf_r2_svgs/).
# Also writes a per-thread-CPU snapshot alongside each SVG.

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_DIR="$(cd "$SCRIPT_DIR/.." && pwd)"
OUT_DIR="${OUT_DIR:-$SCRIPT_DIR/perf_r2_svgs}"
mkdir -p "$OUT_DIR"

AC="$REPO_DIR/target/release/autumn-client"
# Build profiling-enabled PS once
PS_PROF_BIN="$REPO_DIR/target/release/autumn-ps-prof"
if [[ ! -x "$PS_PROF_BIN" ]] || [[ "$REPO_DIR/crates/server/src/bin/partition_server.rs" -nt "$PS_PROF_BIN" ]]; then
    echo "[R2] building profiling-enabled autumn-ps..."
    cd "$REPO_DIR"
    cargo build --bin autumn-ps --release -p autumn-server --features profiling
    cp "$REPO_DIR/target/release/autumn-ps" "$PS_PROF_BIN"
fi

run_one_capture() {
    local label="$1" partitions="$2" thread_filter="$3" duration="${4:-15}"
    local svg="$OUT_DIR/perf_r2_${label}.svg"
    local cpusnap="$OUT_DIR/perf_r2_${label}.thread_cpu.txt"
    local logpath="$OUT_DIR/perf_r2_${label}.log"

    echo "[R2] capture $label  (partitions=$partitions, filter=${thread_filter:-<none>}, duration=${duration}s)"

    export AUTUMN_DATA_ROOT=/dev/shm/autumn-rs
    if (( partitions > 1 )); then
        export AUTUMN_BOOTSTRAP_PRESPLIT="${partitions}:hexstring"
    else
        unset AUTUMN_BOOTSTRAP_PRESPLIT
    fi

    pushd "$REPO_DIR" >/dev/null
    ./cluster.sh clean >/dev/null
    ./cluster.sh reset 3 >/dev/null 2>&1
    # cluster.sh launched the *default* autumn-ps; we need to replace it with the profiling build.
    local default_pid
    default_pid="$(cat "$AUTUMN_DATA_ROOT/pids/ps.pid")"
    kill "$default_pid" 2>/dev/null || true
    # Wait for port release
    for _ in $(seq 1 20); do
        if ! nc -z 127.0.0.1 9201 2>/dev/null; then break; fi
        sleep 0.2
    done

    # Launch profiling-enabled PS
    local pprof_out="$svg"
    AUTUMN_PPROF_SECS="$duration" \
    AUTUMN_PPROF_OUT="$pprof_out" \
    AUTUMN_PPROF_THREADS="$thread_filter" \
    "$PS_PROF_BIN" \
        --psid 1 --port 9201 --manager 127.0.0.1:9001 --advertise 127.0.0.1:9201 \
        > "$logpath" 2>&1 &
    local ps_pid=$!
    echo "[R2]   new PS pid=$ps_pid"

    # Wait for PS to bind
    for _ in $(seq 1 60); do
        if nc -z 127.0.0.1 9201 2>/dev/null; then break; fi
        sleep 0.2
    done
```

**Algorithm sketch:**

1. `lease(size) -> Option<(offset, end)>`: lock; if `in_flight >= depth_cap` return None (caller awaits a semaphore to retry); else `offset = tail; tail += size; in_flight += 1`; unlock. Return (offset, end).
2. `commit_ack(offset, end)`: lock; `pending.insert(offset, end); in_flight -= 1`. Drain pending entries whose `offset == commit` and advance commit to that end. Unlock. Notify semaphore.
3. `rewind_on_error(offset, size)`: lock; if `offset == tail - size` (we are the most recent lease), rewind tail. If we're mid-sequence, mark as poisoned and force caller to seal + alloc new extent (matches existing error handling).

```rust
#[derive(Debug)]
struct StreamAppendState {
    /// Next offset to lease for a new `append_batch`.
    tail: u32,
    /// Highest acked end that forms a contiguous committed prefix.
    commit: u32,
    /// Count of leases that have not yet acked.
    in_flight: u32,
    /// Acked batches whose offset > commit (cannot advance commit yet).
    pending: std::collections::BTreeMap<u32, u32>,
    /// Pipeline cap (read once from env at construction time).
    depth_cap: u32,
}

impl StreamAppendState {
    fn new(depth_cap: u32) -> Self {
        Self { tail: 0, commit: 0, in_flight: 0, pending: Default::default(), depth_cap }
    }
    /// Try to lease an offset range of `size` bytes. Returns `None` if at cap.
    fn try_lease(&mut self, size: u32) -> Option<(u32, u32)> {
        if self.in_flight >= self.depth_cap { return None; }
        let offset = self.tail;
        let end = offset + size;
        self.tail = end;
        self.in_flight += 1;
        Some((offset, end))
    }
    /// Record an ack for the lease `(offset, end)`. Advance commit if prefix contiguous.
    fn ack(&mut self, offset: u32, end: u32) {
        self.pending.insert(offset, end);
        self.in_flight = self.in_flight.saturating_sub(1);
        while let Some((&off, &end2)) = self.pending.iter().next() {
            if off == self.commit {
                self.commit = end2;
                self.pending.remove(&off);
            } else {
                break;
            }
        }
    }
    /// Error rewind: best effort undo. Only safe if this was the most-recent lease.
    fn rewind(&mut self, offset: u32, size: u32) {
        self.in_flight = self.in_flight.saturating_sub(1);
        if offset + size == self.tail {
            self.tail = offset;
        } else {
            // Not safe to rewind mid-sequence; caller must seal + alloc new extent.
        }
    }
}
```

```rust
pub async fn append_batch(
    &self,
    stream_id: u64,
    blocks: &[bytes::Bytes],
    must_sync: bool,
) -> Result<AppendResult> {
    let slot = self.stream_slot(stream_id).await?;
    // 1. Acquire one pipeline permit.
    let _permit = slot.semaphore.acquire().await;

    // 2. Lease an offset range under the state lock.
    let total_size: u32 = blocks.iter().map(|b| b.len() as u32).sum();
    let (offset, end) = {
        let mut state = slot.state.lock().await;
        // The try_lease will always succeed here (semaphore guarantees capacity).
        state.try_lease(total_size).expect("semaphore and lease disagree")
    };

    // 3. Compute replica commit_length (once, under state lock drop).
    let commit = {
        let state = slot.state.lock().await;
        state.commit
    };

    // 4. Fanout to replicas with header.commit = commit, header.start = offset.
    let fanout_result = self.fanout_append(stream_id, offset, end, blocks, must_sync, commit).await;

    // 5. Ack or rewind under state lock.
    let mut state = slot.state.lock().await;
    match &fanout_result {
        Ok(_) => state.ack(offset, end),
        Err(_) => state.rewind(offset, total_size),
    }
    drop(state);
    fanout_result.map(|_| AppendResult { extent_id: self.current_extent(stream_id).await?, offset, end })
}
```

```rust
//! Leader-follower write-batch builder — RocksDB WriteImpl analog.
//!
//! Each partition owns one WriteBatchBuilder. Conn workers push their
//! `(WriteRequest, completion_waker)` into a shared slab guarded by a
//! short spin-ish lock. The P-log thread's write loop periodically
//! drains the slab (on every iteration OR when memtable rotate is
//! needed). After one drain, P-log is the "leader" for that batch:
//! it runs Phase 1/2/3 for all queued requests, then signals all
//! followers' wakers via an `AtomicU64` seq + `AtomicWaker`.
//!
//! This replaces the mpsc + per-request oneshot pattern with:
//!   push (spinlock)  →  drain (once per batch)  →  broadcast (shared waker)
//!
//! Invariants:
//!   1. Slab order preserved (FIFO within a partition).
//!   2. Waker rises only after leader's Phase 3 memtable insert is done.
```

Continue with the struct + impl:

```rust
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;
use parking_lot::Mutex;

pub(crate) struct WriteBatchBuilder {
    queue: Mutex<Vec<super::WriteRequest>>,
    last_seq: AtomicU64,
    waker: futures::task::AtomicWaker,
}

impl WriteBatchBuilder {
    pub(crate) fn new() -> Self {
        Self { queue: Mutex::new(Vec::new()), last_seq: AtomicU64::new(0), waker: Default::default() }
    }
    /// Conn worker pushes request; returns a Completion future.
    pub(crate) fn push(self: &Arc<Self>, req: super::WriteRequest) -> Completion {
        let my_seq = {
            let mut q = self.queue.lock();
            q.push(req);
            q.len() as u64
        };
        Completion { seq: my_seq, builder: self.clone() }
    }
    /// Leader drains all queued requests. Called by P-log write loop.
    pub(crate) fn drain(&self) -> Vec<super::WriteRequest> {
        std::mem::take(&mut *self.queue.lock())
    }
    /// Leader signals all followers that work through `up_to_seq` is done.
    pub(crate) fn signal_complete(&self, up_to_seq: u64) {
        self.last_seq.store(up_to_seq, Ordering::Release);
        self.waker.wake();
    }
}

pub(crate) struct Completion {
    seq: u64,
    builder: Arc<WriteBatchBuilder>,
}

impl std::future::Future for Completion {
    type Output = ();
    fn poll(self: std::pin::Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> std::task::Poll<()> {
        if self.builder.last_seq.load(Ordering::Acquire) >= self.seq {
            std::task::Poll::Ready(())
        } else {
            self.builder.waker.register(cx.waker());
            if self.builder.last_seq.load(Ordering::Acquire) >= self.seq {
                std::task::Poll::Ready(())
            } else {
                std::task::Poll::Pending
            }
        }
    }
}
```
