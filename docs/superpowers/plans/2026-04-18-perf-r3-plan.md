```rust
//! Per-process connection pool for extent nodes (single-threaded compio).
//!
//! One `Rc<RpcClient>` per SocketAddr. Since autumn-rpc's `RpcClient`
//! handles concurrent `call`/`call_vectored` via an internal
//! `Mutex<WriteHalf>` (byte-level frame serialization) plus a background
//! reader that dispatches responses by `req_id`, multiple tasks can
//! drive appends against the same addr simultaneously — true TCP
//! multiplex, not one-caller-at-a-time.
//!
//! Historical note: before R3, this pool used a take/put `RefCell<Option<RpcConn>>`
//! pattern with an embedded sequential `RpcConn` struct. That prevented
//! multiplex even though autumn-rpc supports it at the wire level. R3
//! replaces the custom sequential conn with `Rc<RpcClient>` (shared).

use std::cell::RefCell;
use std::collections::HashMap;
use std::net::SocketAddr;
use std::rc::Rc;
use std::time::Duration;

use anyhow::{anyhow, Result};
use autumn_rpc::client::RpcClient;
use bytes::Bytes;

pub struct ConnPool {
    clients: RefCell<HashMap<SocketAddr, Rc<RpcClient>>>,
}

impl ConnPool {
    pub fn new() -> Self {
        Self {
            clients: RefCell::new(HashMap::new()),
        }
    }

    /// Get or open an RpcClient for `addr`. Uses the existing pool entry
    /// if present; otherwise connects and stashes. Connection is shared
    /// across concurrent callers.
    async fn get_client(&self, addr: SocketAddr) -> Result<Rc<RpcClient>> {
        if let Some(client) = self.clients.borrow().get(&addr).cloned() {
            return Ok(client);
        }
        let client = RpcClient::connect(addr)
            .await
            .map_err(|e| anyhow!("connect {}: {}", addr, e))?;
        self.clients.borrow_mut().insert(addr, client.clone());
        Ok(client)
    }

    /// Evict the pooled client for `addr` (next `get_client` reconnects).
    fn evict(&self, addr: SocketAddr) {
        self.clients.borrow_mut().remove(&addr);
    }

    /// Send an RPC and await the response. On error, evict the client so
    /// the next call reconnects (matches R2 behavior semantically).
    pub async fn call(&self, addr: &str, msg_type: u8, payload: Bytes) -> Result<Bytes> {
        let sock = parse_addr(addr)?;
        let client = self.get_client(sock).await?;
        match client.call(msg_type, payload).await {
            Ok(bytes) => Ok(bytes),
            Err(e) => {
                self.evict(sock);
                Err(anyhow!("{}", e))
            }
        }
    }

    /// Send an RPC with a timeout. Same error-eviction contract as `call`.
    pub async fn call_timeout(
        &self,
        addr: &str,
        msg_type: u8,
        payload: Bytes,
        timeout: Duration,
    ) -> Result<Bytes> {
        let sock = parse_addr(addr)?;
        let client = self.get_client(sock).await?;
        match client.call_timeout(msg_type, payload, timeout).await {
            Ok(bytes) => Ok(bytes),
            Err(e) => {
                self.evict(sock);
                Err(anyhow!("{}", e))
            }
        }
    }

    /// Send an RPC with payload already split into parts (zero-copy).
    pub async fn call_vectored(
        &self,
        addr: &str,
        msg_type: u8,
        payload_parts: Vec<Bytes>,
    ) -> Result<Bytes> {
        let sock = parse_addr(addr)?;
        let client = self.get_client(sock).await?;
        match client.call_vectored(msg_type, payload_parts).await {
            Ok(bytes) => Ok(bytes),
            Err(e) => {
                self.evict(sock);
                Err(anyhow!("{}", e))
            }
        }
    }

    pub fn is_healthy(&self, addr: &str) -> bool {
        let Ok(sock) = parse_addr(addr) else {
            return false;
        };
        self.clients.borrow().contains_key(&sock)
    }
}

impl Default for ConnPool {
    fn default() -> Self {
        Self::new()
    }
}

/// Parse a "host:port" address into a SocketAddr.
pub fn parse_addr(addr: &str) -> Result<SocketAddr> {
    let stripped = addr
        .trim_start_matches("http://")
        .trim_start_matches("https://");
    stripped
        .parse::<SocketAddr>()
        .map_err(|e| anyhow!("invalid address {:?}: {}", addr, e))
}

/// Normalize an address string by stripping any http:// prefix.
pub fn normalize_endpoint(addr: &str) -> String {
    addr.trim_start_matches("http://")
        .trim_start_matches("https://")
        .to_string()
}
```

```rust
        addr: &str,
        msg_type: u8,
        payload: Bytes,
        timeout: Duration,
    ) -> Result<Bytes> {
        use futures::FutureExt;
```

```rust
/// Per-stream append state: tail info, commit cache, and R3 pipeline state.
///
/// R3 state machine fields (`lease_cursor`, `pending_acks`, `in_flight`,
/// `poisoned`) track in-flight appends so the state Mutex can be held
/// only for offset-lease + ack, not across the 3-replica fanout.
///
/// - `tail`: cached `StreamTail` (extent id + replica addrs + eversion).
/// - `commit`: highest acked `end` that forms a contiguous prefix (matches
///             Go's `sc.end`; also serves as `header.commit` for appends
///             that are strictly serialized, i.e. R2 behavior).
/// - `lease_cursor`: next offset to lease. Advances monotonically at lease
///                   time; not rewound on success. Equals `commit` when
///                   `in_flight == 0 && pending_acks.is_empty()`.
/// - `pending_acks`: acked-but-not-yet-prefix batches (offset → end).
/// - `in_flight`: count of leased-but-not-acked batches.
/// - `poisoned`: set on mid-sequence error; causes the next caller to
///               trigger `alloc_new_extent` via an NotFound-equivalent
///               error path.
struct StreamAppendState {
    tail: Option<StreamTail>,
    commit: u32,
    lease_cursor: u32,
    pending_acks: std::collections::BTreeMap<u32, u32>,
    in_flight: u32,
    poisoned: bool,
}

impl StreamAppendState {
    fn new() -> Self {
        Self {
            tail: None,
            commit: 0,
            lease_cursor: 0,
            pending_acks: std::collections::BTreeMap::new(),
            in_flight: 0,
            poisoned: false,
        }
    }

    /// Reset state for a newly-allocated extent. Called under the state
    /// Mutex whenever `alloc_new_extent` returns a fresh extent. Preserves
    /// `tail` (caller will overwrite it separately).
    fn reset_for_new_extent(&mut self) {
        self.commit = 0;
        self.lease_cursor = 0;
        self.pending_acks.clear();
        self.in_flight = 0;
        self.poisoned = false;
    }

    /// Lease an offset range for a new append. Caller must hold the state
    /// Mutex. Always succeeds; the in_flight counter tracks outstanding
    /// leases for `rewind_on_error` and for R3 integration tests.
    fn lease(&mut self, size: u32) -> (u32, u32) {
        let offset = self.lease_cursor;
        let end = offset + size;
        self.lease_cursor = end;
        self.in_flight += 1;
        (offset, end)
    }

    /// Record a successful ack: insert into pending, then advance commit
    /// through any contiguous prefix.
    fn ack(&mut self, offset: u32, end: u32) {
        self.pending_acks.insert(offset, end);
        self.in_flight = self.in_flight.saturating_sub(1);
        while let Some((&off, &end_of_slot)) = self.pending_acks.iter().next() {
            if off == self.commit {
                self.commit = end_of_slot;
                self.pending_acks.remove(&off);
            } else {
                break;
            }
        }
    }

    /// Record a failed ack. If this was the most-recently-leased batch
    /// (no newer leases outstanding), rewind tail. Otherwise, poison the
    /// stream so the next caller's append will force a new extent alloc.
    fn rewind_or_poison(&mut self, offset: u32, size: u32) {
        self.in_flight = self.in_flight.saturating_sub(1);
        if offset + size == self.lease_cursor {
            // Most-recent lease: it's safe to rewind.
            self.lease_cursor = offset;
        } else {
            // Newer leases are in flight; we can't rewind without gaps.
            self.poisoned = true;
        }
    }
}
```

## Task 5 · Rewrite `append_payload_segments` to use the lease state machine

### Step 1: Read the existing function to understand its shape

Key behaviors to preserve (from the R2 code):

- `revision` passed into the AppendReq header.
- Auto-load `state.tail` from manager if `None`.
- Query `current_commit(&tail)` on first append to an existing extent (when `state.commit == 0 && state.tail.is_some()`).
- 3-replica parallel fanout via `join_all`.
- Collect results: OK → extract `(offset, end)`, first ok result wins. If any `CODE_NOT_FOUND` → mark and retry with alloc_new_extent. Other errors → retry (up to `MAX_ALLOC_PER_APPEND = 3`).
- `MAX_ALLOC_PER_APPEND` caps total alloc attempts across a single append call.
- `state.commit = appended.end` after a successful retry-exit.

The R3 rewrite preserves all of these while changing WHERE the mutex is held. The MAX_ALLOC_PER_APPEND + NotFound retry loop stays inside the "held" region (lease + fanout writes under mutex) for simplicity: retrying while holding the mutex is still valid because retries happen rarely (only on NotFound or error) and don't add contention in the hot path.

Actually, the cleanest split is:
1. **Lock mutex** for lease + allocate new extent if poisoned + build 3 fanout RPC send-frames (bytes hit 3 sockets).
2. **Drop mutex.** Await the 3 `rx.await`s in `join_all`.
3. **Lock mutex** for ack or rewind, handle retry-on-NotFound / alloc_new_extent.

Retries stay inside the locked region because each retry is lease-of-a-new-offset + send-frames. Let's go with that.

   ```rust
            }
   ```
```rust
        }
```

### Step 6: Rewrite `append_payload_segments`

Now back in `crates/stream/src/client.rs:454`. Replace the body of `async fn append_payload_segments` (preserving signature) with logic matching the R3 algorithm. Show the key region here; the implementer preserves all surrounding imports and helper calls (e.g., `load_stream_tail`, `current_commit`, `alloc_new_extent`, `AppendResult` construction).

**Illustrative skeleton of the new body** (implementer adapts to the exact existing structure):

```rust
    async fn append_payload_segments(
        &self,
        stream_id: u64,
        segments: Vec<Bytes>,
        must_sync: bool,
    ) -> Result<AppendResult> {
        let payload_len_u32: u32 = segments.iter().map(|s| s.len() as u32).sum();
        let append_started_at = Instant::now();
        let state_arc = self.stream_state(stream_id);

        let segments = Arc::new(segments);

        let mut retry = 0usize;
        let mut alloc_count = 0u32;
        const MAX_ALLOC_PER_APPEND: u32 = 3;
        let mut extent_lookup_elapsed = Duration::default();
        let mut fanout_elapsed = Duration::default();
        let mut lock_wait_total = Duration::default();

        loop {
            // ── Phase 1: short lock for lease + pre-send sanity ──
            let lock_started_at = Instant::now();
            let mut state = state_arc.lock().await;
            lock_wait_total += lock_started_at.elapsed();

            // Handle poisoned stream: force alloc_new_extent path (mirror
            // the existing NotFound handling).
            if state.poisoned {
                state.poisoned = false;
                state.reset_for_new_extent();
                state.tail = None;  // force reload
                // fall through: loop iter will hit the None branch below.
            }

            // Load tail if cached None.
            let tail = match &state.tail {
                Some(t) => t.clone(),
                None => {
                    // Release lock for the alloc/fetch (preserve existing pattern).
                    drop(state);
                    if alloc_count >= MAX_ALLOC_PER_APPEND {
                        return Err(anyhow!("exceeded MAX_ALLOC_PER_APPEND"));
                    }
                    alloc_count += 1;
                    let t = self.load_or_alloc_stream_tail(stream_id).await?;
                    let mut state = state_arc.lock().await;
                    state.tail = Some(t.clone());
                    state.reset_for_new_extent();
                    drop(state);
                    continue;
                }
            };

            let revision = self.revision;
            let commit = if state.commit == 0 && state.tail.is_some() && state.lease_cursor == 0 {
                let c = self.current_commit(&tail).await.unwrap_or(0);
                state.commit = c;
                state.lease_cursor = c;
                c
            } else {
                state.commit
            };

            // Lease.
            let (offset, end) = state.lease(payload_len_u32);

            // Header.commit MUST be the lease-time cursor (= offset, which equals
            // the end of the previous lease). Per spec §5.2 Option A. This
            // differs from R2 where header.commit was state.commit.
            let header_commit = offset;

            let extent_id = tail.extent.extent_id;
            let append_hdr = AppendReq::encode_header(
                extent_id,
                tail.extent.eversion,
                header_commit,
                revision,
                must_sync,
            );

            // Build vectored parts once (cloneable Bytes for the 3 sends).
            let build_parts = || -> Vec<Bytes> {
                let mut parts: Vec<Bytes> = Vec::with_capacity(1 + segments.len());
                parts.push(append_hdr.clone());
                for seg in segments.iter() {
                    parts.push(seg.clone());
                }
                parts
            };

            // Fire 3 send_vectored's WHILE HOLDING the lock (bytes hit sockets
            // in lease order). Collect the oneshot receivers.
            let extent_lookup_started_at = Instant::now();
            extent_lookup_elapsed += extent_lookup_started_at.elapsed();

            let fanout_started_at = Instant::now();
            let mut receivers: Vec<(String, Result<oneshot::Receiver<autumn_rpc::Frame>>)> =
                Vec::with_capacity(tail.replica_addrs.len());
            for addr in &tail.replica_addrs {
                let rx_res = self
                    .pool
                    .send_vectored(addr, MSG_APPEND, build_parts())
                    .await;
                receivers.push((addr.clone(), rx_res));
            }

            // ── Drop the lock — the expensive await happens outside. ──
            drop(state);

            // Await all 3 responses concurrently.
            let wait_futs = receivers.into_iter().map(|(addr, rx_res)| async move {
                match rx_res {
                    Err(e) => (addr, Err(e)),
                    Ok(rx) => match rx.await {
                        Err(_) => (addr, Err(anyhow!("rpc connection closed"))),
                        Ok(frame) => {
                            if frame.is_error() {
                                let (code, msg) = autumn_rpc::RpcError::decode_status(&frame.payload);
                                (addr, Err(anyhow!("rpc error ({:?}): {}", code, msg)))
                            } else {
                                (addr, Ok(frame.payload))
                            }
                        }
                    },
                }
            });
            let results = join_all(wait_futs).await;
            fanout_elapsed += fanout_started_at.elapsed();

            // Parse results: look for NotFound or other errors.
            let mut first_resp: Option<AppendResp> = None;
            let mut saw_not_found = false;
            let mut append_error: Option<anyhow::Error> = None;

            for (addr, result) in results {
                let resp_bytes = match result {
                    Ok(b) => b,
                    Err(e) => {
                        append_error = Some(e);
                        break;
                    }
                };
                let inner = match AppendResp::decode(resp_bytes) {
                    Ok(r) => r,
                    Err(e) => {
                        append_error = Some(anyhow!("decode AppendResp from {addr}: {e}"));
                        break;
                    }
                };
                if inner.code == CODE_NOT_FOUND {
                    saw_not_found = true;
                    continue;
                }
                if inner.code != CODE_OK {
                    append_error = Some(anyhow!(
                        "append replica {} returned code {}: {}",
                        addr, inner.code, inner.message
                    ));
                    break;
                }
                if first_resp.is_none() {
                    first_resp = Some(inner);
                }
            }

            // ── Phase 3: short lock for ack or rewind ──
            let mut state = state_arc.lock().await;

            if let Some(resp) = first_resp {
                // Success: ack and return.
                state.ack(offset, end);
                let record_duration = append_started_at.elapsed();
                self.metrics.record(
                    &self.owner_key,
                    lock_wait_total,
                    extent_lookup_elapsed,
                    fanout_elapsed,
                    record_duration,
                    retry as u64,
                );
                return Ok(AppendResult {
                    extent_id,
                    offset: resp.offset,
                    end: resp.end,
                });
            }

            // Failure path: rewind or poison.
            state.rewind_or_poison(offset, payload_len_u32);

            if saw_not_found {
                // Sealed / replaced extent — seal locally and alloc new.
                state.reset_for_new_extent();
                state.tail = None;
                if alloc_count >= MAX_ALLOC_PER_APPEND {
                    return Err(anyhow!("exceeded MAX_ALLOC_PER_APPEND"));
                }
                alloc_count += 1;
                drop(state);
                retry += 1;
                continue;
            }

            // Generic error: retry up to budget.
            if let Some(e) = append_error {
                if retry < 2 {
                    retry += 1;
                    drop(state);
                    compio::time::sleep(Duration::from_millis(100)).await;
                    continue;
                }
                // After 2 retries, force a new extent like the existing code does.
                state.reset_for_new_extent();
                state.tail = None;
                if alloc_count >= MAX_ALLOC_PER_APPEND {
                    return Err(anyhow!("exceeded MAX_ALLOC_PER_APPEND: {e}"));
                }
                alloc_count += 1;
                drop(state);
                retry = 0;
                continue;
            }

            // Unreachable.
            return Err(anyhow!("append: no response and no error"));
        }
    }
```

### Step 2: Add `header_commit_is_lease_time_cursor` to `pipeline_tests`

This test exercises the core correctness invariant from spec §5.2 (Option A). It verifies that `header.commit(N)` equals the lease-time cursor, NOT the ack-time `state.commit`.

The test uses the pure state machine (no RPC), inspecting what values `append_payload_segments` WOULD send.

```rust
    #[test]
    fn header_commit_is_lease_time_cursor() {
        // Simulate the sequence of leases and compute what header.commit
        // would be for each batch. We verify:
        //   header.commit(N) == offset(N) == end(N-1)
        // and specifically that it's NOT state.commit (which trails).
        let mut state = StreamAppendState::new();

        // Batch 0: lease, don't ack yet.
        let header_commit_0 = state.lease_cursor;   // what append_payload_segments captures
        let (o0, e0) = state.lease(100);
        assert_eq!(header_commit_0, 0);
        assert_eq!(o0, 0);
        assert_eq!(e0, 100);

        // Batch 1: lease BEFORE batch 0 acks. header.commit(1) must be end(0) = 100.
        let header_commit_1 = state.lease_cursor;
        let (o1, _e1) = state.lease(200);
        assert_eq!(header_commit_1, 100,
            "BUG: header.commit(1) == {} but must == end(0) == 100",
            header_commit_1);
        assert_eq!(state.commit, 0,
            "state.commit is still 0 (no acks yet) — this would be WRONG as header.commit");

        // Ack batch 1 out of order.
        state.ack(o1, 300);
        assert_eq!(state.commit, 0, "ack(b1) alone does not advance commit (b0 still in-flight)");

        // Batch 2: header.commit(2) must equal state.lease_cursor = 300 (= end(1)).
        let header_commit_2 = state.lease_cursor;
        assert_eq!(header_commit_2, 300);

        // Finally ack batch 0 — now commit advances through the whole prefix.
        state.ack(o0, e0);
        assert_eq!(state.commit, 300);
    }
```

```rust
    #[compio::test]
    async fn concurrent_lease_order_is_monotonic() {
        // 4 concurrent leases; verify returned (offset, end) pairs don't overlap
        // and that the sum of sizes == final lease_cursor.
        use std::sync::Mutex as SyncMutex;
        let state = std::sync::Arc::new(SyncMutex::new(StreamAppendState::new()));
        let leases = std::sync::Arc::new(SyncMutex::new(Vec::<(u32, u32)>::new()));

        let mut tasks = Vec::new();
        for i in 0..4u32 {
            let state = state.clone();
            let leases = leases.clone();
            let size = 100 + i * 10;
            tasks.push(std::thread::spawn(move || {
                // Take the lock and lease atomically.
                let mut s = state.lock().unwrap();
                let lease = s.lease(size);
                leases.lock().unwrap().push(lease);
            }));
        }
        for t in tasks { t.join().unwrap(); }

        let mut ls = leases.lock().unwrap().clone();
        ls.sort();
        for i in 1..ls.len() {
            assert!(
                ls[i].0 >= ls[i-1].1,
                "leases overlap: {:?} vs {:?}",
                ls[i-1], ls[i]
            );
        }
        let s = state.lock().unwrap();
        let total_size: u32 = (0..4).map(|i| 100 + i*10).sum();
        assert_eq!(s.lease_cursor, total_size);
        assert_eq!(s.in_flight, 4);
    }
```

Note this test covers the invariant that MATTERS for conn-ordering (contiguous non-overlapping leases under a mutex). The full conn-level ordering test (frames arrive at ExtentNode in lease order) is validated end-to-end by the integration test in Task 7, not here.

```rust
//! R3 integration test: pipelined writes survive PS kill + restart.
//!
//! Validates spec §5.5 (Crash Recovery) under R3's pipelined-append
//! state machine. Reuses the test-support module from R1 F062.

mod support;

use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;
use std::time::Duration;

use support::{
    poll_until_async, ps_get, ps_put, setup_full_partition, ShutdownFlag,
};

#[compio::test]
async fn r3_pipelined_writes_survive_ps_restart() {
    let shutdown = ShutdownFlag::new();
    let infra = setup_full_partition(&shutdown).await;

    // Fire 256 concurrent writers against the PS; each repeatedly puts
    // `pc_<writer_id>_<seq>` keys with 4 KB values. Collect the set of
    // (key, value) pairs for which PUT returned Ok — these must survive
    // the restart.
    let acked: Arc<parking_lot::Mutex<std::collections::HashMap<String, Vec<u8>>>> =
        Arc::new(parking_lot::Mutex::new(std::collections::HashMap::new()));
    let total_ops = Arc::new(AtomicU64::new(0));
    let writer_shutdown = Arc::new(std::sync::atomic::AtomicBool::new(false));

    let mut writer_handles = Vec::new();
    for wid in 0..16usize {
        let acked = acked.clone();
        let total_ops = total_ops.clone();
        let writer_shutdown = writer_shutdown.clone();
        let ps_addr = infra.ps_addr.clone();
        let part_id = infra.part_id;
        let h = std::thread::spawn(move || {
            compio::runtime::Runtime::new().unwrap().block_on(async move {
                let mut seq = 0u64;
                while !writer_shutdown.load(Ordering::Relaxed) {
                    let key = format!("pc_{wid}_{seq}");
                    let value = vec![b'v'; 4096];
                    seq += 1;
                    match ps_put(&ps_addr, part_id, &key, &value, /*must_sync=*/true).await {
                        Ok(()) => {
                            acked.lock().insert(key, value);
                            total_ops.fetch_add(1, Ordering::Relaxed);
                        }
                        Err(_) => {
                            // Connection broke during kill — expected. Retry on new PS.
                            compio::time::sleep(Duration::from_millis(50)).await;
                        }
                    }
                }
            });
        });
        writer_handles.push(h);
    }

    // Let writes flow for 2 seconds.
    compio::time::sleep(Duration::from_secs(2)).await;

    let ops_before_kill = total_ops.load(Ordering::Relaxed);
    assert!(
        ops_before_kill > 1_000,
        "expected > 1k ops before kill, got {}",
        ops_before_kill
    );

    // Kill the PS.
    infra.kill_ps();

    // Start a fresh PS with the same psid.
    let new_infra = infra.restart_ps_with_same_id().await;

    // Wait for GetRegions to show partition served by the new PS.
    poll_until_async(Duration::from_secs(30), || async {
        new_infra.get_regions().await
            .map(|r| r.contains_partition(new_infra.part_id))
            .unwrap_or(false)
    }).await.expect("partition re-assigned to new PS");

    // Let writers keep going against the new PS for another second.
    compio::time::sleep(Duration::from_secs(1)).await;
    writer_shutdown.store(true, Ordering::Relaxed);
    for h in writer_handles { h.join().unwrap(); }

    // Read back every acked key. Every one must be present AND match value.
    let acked = acked.lock().clone();
    assert!(!acked.is_empty(), "no writes acked — test setup wrong");
    let mut verified = 0usize;
    for (key, expected_value) in &acked {
        let actual = ps_get(&new_infra.ps_addr, new_infra.part_id, key).await
            .unwrap_or_else(|e| panic!("get {key} after restart: {e}"));
        assert_eq!(
            &actual, expected_value,
            "value mismatch for acked key {key}"
        );
        verified += 1;
    }
    println!("R3 restart test: {} acked keys all readable and correct", verified);

    shutdown.set();
    new_infra.teardown().await;
}
```
